```App.tsx
import React, { useState } from 'react'
import { AnimatePresence } from 'framer-motion'
import { BottomNav, PageType } from './components/BottomNav'
import { HomePage } from './pages/HomePage'
import { BookingPage } from './pages/BookingPage'
import { QueuePage } from './pages/QueuePage'
import { ProfilePage } from './pages/ProfilePage'
export function App() {
  const [activePage, setActivePage] = useState<PageType>('home')
  const renderPage = () => {
    switch (activePage) {
      case 'home':
        return <HomePage key="home" onBookClick={() => setActivePage('book')} />
      case 'book':
        return <BookingPage key="book" />
      case 'queue':
        return <QueuePage key="queue" />
      case 'profile':
        return <ProfilePage key="profile" />
      default:
        return <HomePage key="home" onBookClick={() => setActivePage('book')} />
    }
  }
  return (
    <div className="min-h-screen bg-luxury-black text-luxury-white font-inter relative overflow-hidden bg-noise">
      {/* Mobile-first container */}
      <div className="max-w-md mx-auto min-h-screen relative shadow-2xl bg-luxury-black">
        <AnimatePresence mode="wait">{renderPage()}</AnimatePresence>

        <BottomNav activePage={activePage} onChange={setActivePage} />
      </div>
    </div>
  )
}
```

```components/BookingCalendar.tsx
import React from 'react'
import { motion } from 'framer-motion'
import { ChevronLeft, ChevronRight } from 'lucide-react'
interface BookingCalendarProps {
  selectedDate: number | null
  onSelectDate: (date: number) => void
}
export function BookingCalendar({
  selectedDate,
  onSelectDate,
}: BookingCalendarProps) {
  const days = ['S', 'M', 'T', 'W', 'T', 'F', 'S']
  // Mock current month: 30 days, starts on Tuesday (index 2)
  const blanks = Array(2).fill(null)
  const dates = Array.from(
    {
      length: 30,
    },
    (_, i) => i + 1,
  )
  const today = 15 // Mock today
  return (
    <div className="w-full mb-10">
      <div className="flex justify-between items-center mb-6 px-2">
        <button className="p-2 text-luxury-muted hover:text-luxury-white transition-colors">
          <ChevronLeft className="w-5 h-5" />
        </button>
        <span className="font-playfair text-lg text-luxury-white">
          October 2023
        </span>
        <button className="p-2 text-luxury-muted hover:text-luxury-white transition-colors">
          <ChevronRight className="w-5 h-5" />
        </button>
      </div>

      <div className="grid grid-cols-7 gap-y-6 gap-x-2 text-center">
        {days.map((day, i) => (
          <div
            key={`day-${i}`}
            className="font-inter text-xs text-luxury-muted font-medium"
          >
            {day}
          </div>
        ))}

        {blanks.map((_, i) => (
          <div key={`blank-${i}`} className="h-10" />
        ))}

        {dates.map((date) => {
          const isPast = date < today
          const isSelected = selectedDate === date
          return (
            <div key={date} className="flex justify-center items-center h-10">
              <motion.button
                whileTap={
                  !isPast
                    ? {
                        scale: 0.9,
                      }
                    : {}
                }
                onClick={() => !isPast && onSelectDate(date)}
                disabled={isPast}
                className={`
                  w-10 h-10 rounded-full flex items-center justify-center font-inter text-sm transition-colors duration-300
                  ${isPast ? 'text-luxury-white/20 cursor-not-allowed' : 'text-luxury-white'}
                  ${isSelected ? 'bg-luxury-champagne text-luxury-black font-semibold' : ''}
                  ${!isPast && !isSelected ? 'hover:bg-luxury-green/50' : ''}
                `}
              >
                {date}
              </motion.button>
            </div>
          )
        })}
      </div>
    </div>
  )
}
```

```components/BottomNav.tsx
      label: 'PROFILE',
```
